How to use Pluto.js

Input Datetime


var year = 2000;
var month = 1; // January
var day = 15;
var hours = 21; // 9 p.m.
var minutes = 59;
var seconds = 59;
var timezone = 9; // Eastern hemisphere is plus

$pl.setDate(year, month, day, hours, minutes, seconds, timezone);
    

var dateArray = {
  year : 2000,
  month : 1,
  day : 15,
  hours : 21,
  minutes : 59,
  seconds : 59,
  timezone : 9,
}

$pl.setDateArray(dateArray);
    

var julianDate = 2345678;

$pl.setJulDay(julianDate);
